// ========================================================
// API Interfaces & Retrofit Instance
// ========================================================
interface RawgApiService {
    @GET("games")
    suspend fun getPopularGames(
        @Query("key") apiKey: String,
        @Query("page") page: Int
    ): GameListResponse

    @GET("games")
    suspend fun getFilteredGames(
        @Query("key") apiKey: String,
        @Query("page") page: Int,
        @Query("platforms") platforms: String? = null,
        @Query("genres") genres: String? = null,
        @Query("ordering") ordering: String? = null
    ): GameListResponse

    @GET("games/{id}")
    suspend fun getGameDetails(
        @Path("id") gameId: Int,
        @Query("key") apiKey: String
    ): GameDetailModel

    @GET("games/{id}/screenshots")
    suspend fun getGameScreenshots(
        @Path("id") gameId: Int,
        @Query("key") apiKey: String
    ): ScreenshotResponse

    @GET("games/{id}/movies")
    suspend fun getGameTrailers(
        @Path("id") gameId: Int,
        @Query("key") apiKey: String
    ): TrailerResponse
}

        .addConverterFactory(GsonConverterFactory.create())

}

@Composable
fun AppRoot() {
    val auth = FirebaseAuth.getInstance()

@Composable

                Text(if (isLoading) "Creating Account..." else "Register & Start Journey 🎮", color = Color.Black, fontWeight = FontWeight.Bold)

            TextButton(onClick = { keyboardController?.hide(); focusManager.clearFocus(); onNavigateToLogin() }) {
                Text("Already have an account? Log In", color = LudexAccent)

@Composable
fun MainDashboard(
    filter: Pair<String, String>?,
    onClearFilter: () -> Unit,
    onGameClick: (GameModel) -> Unit,
    onDiscoverMoreClick: () -> Unit
) {
    val gamesList = remember { mutableStateListOf<GameModel>() }
    var currentPage by remember { mutableStateOf(1) }
    var isLoading by remember { mutableStateOf(false) }
    var isEndReached by remember { mutableStateOf(false) }
    var errorMessage by remember { mutableStateOf<String?>(null) }
    var dragOffset by remember { mutableStateOf(0f) }
    var isRefreshing by remember { mutableStateOf(false) }
    val coroutineScope = rememberCoroutineScope()
    var searchQuery by remember { mutableStateOf("") }

    // مستدعي تركيز وهمي لسحب التركيز تماماً من الـ TextField ومنع ظهوره باللون الأخضر
    val backgroundFocusRequester = remember { FocusRequester() }
    val focusManager = LocalFocusManager.current
    val keyboardController = LocalSoftwareKeyboardController.current

    val fetchNextPage: (Boolean) -> Unit = { refresh ->
        if (!isLoading) {
            isLoading = true
            errorMessage = null
            if (refresh) {
                currentPage = 1; isEndReached = false
            }
            coroutineScope.launch {
                try {
                    if (currentPage == 1 && refresh) delay(200)

                    val response = if (filter != null) {
                        when (filter.first) {
                            "platforms" -> RetrofitInstance.rawgApi.getFilteredGames(apiKey = RAWG_API_KEY, page = currentPage, platforms = filter.second)
                            "genres" -> RetrofitInstance.rawgApi.getFilteredGames(apiKey = RAWG_API_KEY, page = currentPage, genres = filter.second)
                            "ordering" -> RetrofitInstance.rawgApi.getFilteredGames(apiKey = RAWG_API_KEY, page = currentPage, ordering = filter.second)
                            else -> RetrofitInstance.rawgApi.getPopularGames(apiKey = RAWG_API_KEY, page = currentPage)
                        }
                    } else {
                        RetrofitInstance.rawgApi.getFilteredGames(apiKey = RAWG_API_KEY, page = currentPage, ordering = "-relevance,-metacritic")
                    }

                    if (refresh) gamesList.clear()

                    if (response.results.isNotEmpty()) {
                        val validGames = response.results.filter { !it.background_image.isNullOrBlank() }
                        if (filter == null) {
                            gamesList.addAll(validGames.shuffled())
                        } else {
                            gamesList.addAll(validGames)
                        }
                        currentPage++
                    } else {
                        isEndReached = true
                    }
                } catch (e: Exception) {
                    errorMessage = "Connection Lost"
                } finally {
                    isLoading = false; isRefreshing = false; dragOffset = 0f
                }
            }
        }
    }

    LaunchedEffect(filter) {
        gamesList.clear()
        fetchNextPage(true)
    }

    // إجبار التركيز على الانتقال للخلفية فور تشغيل الشاشة وإخفاء الكيبورد بالقوة
    LaunchedEffect(Unit) {
        delay(100)
        backgroundFocusRequester.requestFocus()
        focusManager.clearFocus(force = true)
        keyboardController?.hide()
    }

    val nestedScrollConnection = remember {
        object : NestedScrollConnection {
            override fun onPreScroll(available: Offset, source: NestedScrollSource): Offset {
                if (available.y > 0 && !isRefreshing) {
                    if (dragOffset < 300f) {
                        dragOffset += available.y; return Offset(0f, available.y)
                    }
                }
                if (available.y < 0 && dragOffset > 0f && !isRefreshing) {
                    dragOffset = (dragOffset + available.y).coerceAtLeast(0f)
                    return Offset(0f, available.y)
                }
                return Offset.Zero
            }

            override fun onPostScroll(consumed: Offset, available: Offset, source: NestedScrollSource): Offset {
                if (dragOffset > 150f && !isRefreshing) {
                    isRefreshing = true; fetchNextPage(true)
                } else if (!isRefreshing) {
                    dragOffset = 0f
                }
                return super.onPostScroll(consumed, available, source)
            }
        }
    }

    Box(modifier = Modifier.fillMaxSize().nestedScroll(nestedScrollConnection)) {
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(horizontal = 14.dp)
                .focusRequester(backgroundFocusRequester) // ربط الحاوية بمستدعي التركيز
                .focusable() // جعلها قابلة لاستقبال التركيز
        ) {
            Row(
                modifier = Modifier.fillMaxWidth().padding(vertical = 12.dp),
                horizontalArrangement = Arrangement.SpaceBetween,
                verticalAlignment = Alignment.CenterVertically
            ) {
                Text(text = "LUDEX", color = LudexAccent, fontSize = 22.sp, fontWeight = FontWeight.Black, letterSpacing = 2.sp)

                OutlinedTextField(
                    value = searchQuery,
                    onValueChange = { newValue -> searchQuery = newValue },
                    placeholder = { Text("Type to search...", color = LudexTextSecondary, fontSize = 12.sp) },
                    leadingIcon = { Icon(Icons.Default.Search, contentDescription = null, tint = LudexTextSecondary, modifier = Modifier.size(16.dp)) },
                    singleLine = true,
                    modifier = Modifier.weight(1f).padding(start = 12.dp).height(46.dp),
                    shape = RoundedCornerShape(24.dp),
                    colors = OutlinedTextFieldDefaults.colors(
                        focusedContainerColor = LudexCardBg,
                        unfocusedContainerColor = LudexCardBg,
                        focusedBorderColor = LudexAccent,
                        unfocusedBorderColor = Color.Transparent,
                        focusedTextColor = LudexTextPrimary,
                        unfocusedTextColor = LudexTextPrimary
                    )
                )

                if (filter != null) {
                    IconButton(onClick = onClearFilter) {
                        Icon(Icons.Default.Clear, contentDescription = "Clear Filter", tint = LudexAccent)
                    }
                }
            }

            Spacer(modifier = Modifier.height(16.dp))

            if (gamesList.isEmpty() && isLoading && !isRefreshing) {
                Box(modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center) {
                    CircularProgressIndicator(color = LudexAccent)
                }
            } else if (errorMessage != null) {
                Box(modifier = Modifier.fillMaxSize().padding(24.dp), contentAlignment = Alignment.Center) {
                    Column(horizontalAlignment = Alignment.CenterHorizontally) {
                        Image(
                            painter = painterResource(id = R.drawable.network_art),
                            contentDescription = "No Connection",
                            modifier = Modifier.size(200.dp),
                            contentScale = ContentScale.Fit
                        )
                        Spacer(modifier = Modifier.height(16.dp))
                        Text(text = "Connection Lost. Please try again.", color = LudexTextSecondary, fontSize = 14.sp)
                        Spacer(modifier = Modifier.height(12.dp))
                        Button(onClick = { fetchNextPage(true) }, colors = ButtonDefaults.buttonColors(containerColor = LudexAccent)) {
                            Text("Retry", color = Color.Black, fontWeight = FontWeight.Bold)
                        }
                    }
                }
            } else {
                LazyVerticalGrid(
                    columns = GridCells.Fixed(2),
                    horizontalArrangement = Arrangement.spacedBy(12.dp),
                    verticalArrangement = Arrangement.spacedBy(14.dp),
                    modifier = Modifier.fillMaxSize()
                ) {
                    itemsIndexed(gamesList) { index, game ->
                        if (index >= gamesList.size - 2 && !isLoading && !isEndReached) {
                            LaunchedEffect(Unit) { fetchNextPage(false) }
                        }
                        GameCard(game = game, onClick = { onGameClick(game) })
                    }
                }
            }
        }
    }
}

@Composable
fun GameDetailScreen(gameId: Int, onBack: () -> Unit) {
    var game by remember { mutableStateOf<GameDetailModel?>(null) }
    val screenshots = remember { mutableStateListOf<ScreenshotModel>() }
    var trailer by remember { mutableStateOf<String?>(null) }
    var isLoading by remember { mutableStateOf(true) }
    var errorMessage by remember { mutableStateOf<String?>(null) }

    val reviewsList = remember { mutableStateListOf<UserReview>() }
    var commentText by remember { mutableStateOf("") }
    var userRating by remember { mutableStateOf(5) }

    val db = FirebaseFirestore.getInstance()
    val auth = FirebaseAuth.getInstance()
    val context = LocalContext.current

    LaunchedEffect(gameId) {
        isLoading = true
        errorMessage = null
        try {
            game = RetrofitInstance.rawgApi.getGameDetails(gameId, RAWG_API_KEY)

            try {
                val scrResponse = RetrofitInstance.rawgApi.getGameScreenshots(gameId, RAWG_API_KEY)
                screenshots.clear()
                screenshots.addAll(scrResponse.results)
            } catch(e: Exception) { }

            try {
                val trlResponse = RetrofitInstance.rawgApi.getGameTrailers(gameId, RAWG_API_KEY)
                trailer = trlResponse.results.firstOrNull()?.data?.max
            } catch(e: Exception) { }

            db.collection("Reviews")
                .whereEqualTo("gameId", gameId)
                .addSnapshotListener { snapshot, _ ->
                    if (snapshot != null) {
                        reviewsList.clear()
                        val items = snapshot.toObjects(UserReview::class.java)
                        reviewsList.addAll(items.sortedByDescending { it.timestamp })
                    }
                }
        } catch (e: Exception) {
            errorMessage = "Failed to load game details."
        } finally {
            isLoading = false
        }
    }

    if (isLoading) {
        Box(modifier = Modifier.fillMaxSize().background(LudexBackground), contentAlignment = Alignment.Center) {
            CircularProgressIndicator(color = LudexAccent)
        }
    } else if (errorMessage != null || game == null) {
        Box(modifier = Modifier.fillMaxSize().background(LudexBackground).padding(24.dp), contentAlignment = Alignment.Center) {
            Column(horizontalAlignment = Alignment.CenterHorizontally) {
                IconButton(onClick = onBack, modifier = Modifier.align(Alignment.Start)) {
                    Icon(Icons.AutoMirrored.Filled.ArrowBack, "Back", tint = LudexAccent)
                }
                Spacer(modifier = Modifier.height(20.dp))
                Text(errorMessage ?: "Unknown Error", color = LudexTextSecondary)
            }
        }
    } else {
        val scrollState = rememberScrollState()
        Column(modifier = Modifier.fillMaxSize().background(LudexBackground).verticalScroll(scrollState)) {
            Box(modifier = Modifier.fillMaxWidth().height(260.dp)) {
                AsyncImage(
                    model = game!!.background_image,
                    contentDescription = game!!.name,
                    modifier = Modifier.fillMaxSize(),
                    contentScale = ContentScale.Crop
                )
                Box(modifier = Modifier.fillMaxSize().background(Brush.verticalGradient(colors = listOf(Color.Transparent, LudexBackground), startY = 300f)))

                IconButton(onClick = onBack, modifier = Modifier.padding(16.dp).align(Alignment.TopStart).background(Color.Black.copy(alpha = 0.5f), RoundedCornerShape(50))) {
                    Icon(Icons.AutoMirrored.Filled.ArrowBack, "Back", tint = Color.White)
                }
            }

            Column(modifier = Modifier.padding(horizontal = 16.dp)) {
                Text(text = game!!.name, color = LudexTextPrimary, fontSize = 26.sp, fontWeight = FontWeight.Black)
                Spacer(modifier = Modifier.height(6.dp))
                Row(verticalAlignment = Alignment.CenterVertically) {
                    Icon(Icons.Default.Star, contentDescription = null, tint = LudexAccent, modifier = Modifier.size(18.dp))
                    Spacer(modifier = Modifier.width(4.dp))
                    Text(text = "${game!!.rating} / 5", color = LudexAccent, fontSize = 14.sp, fontWeight = FontWeight.Bold)
                }

                Spacer(modifier = Modifier.height(20.dp))
                Text(text = "ABOUT THE GAME", color = LudexTextPrimary, fontSize = 14.sp, fontWeight = FontWeight.Bold, letterSpacing = 1.sp)
                Spacer(modifier = Modifier.height(6.dp))
                Text(text = game!!.description_raw ?: "No description available.", color = LudexTextSecondary, fontSize = 14.sp, lineHeight = 20.sp)

                if (screenshots.isNotEmpty()) {
                    Spacer(modifier = Modifier.height(24.dp))
                    Text(text = "SCREENSHOTS", color = LudexTextPrimary, fontSize = 14.sp, fontWeight = FontWeight.Bold, letterSpacing = 1.sp)
                    Spacer(modifier = Modifier.height(10.dp))
                    Row(modifier = Modifier.fillMaxWidth().horizontalScroll(rememberScrollState()), horizontalArrangement = Arrangement.spacedBy(10.dp)) {
                        screenshots.forEach { scr ->
                            Card(shape = RoundedCornerShape(8.dp), modifier = Modifier.size(width = 200.dp, height = 120.dp)) {
                                AsyncImage(model = scr.image, contentDescription = null, modifier = Modifier.fillMaxSize(), contentScale = ContentScale.Crop)
                            }
                        }
                    }
                }

                if (trailer != null) {
                    Spacer(modifier = Modifier.height(24.dp))
                    Text(text = "GAME TRAILER", color = LudexTextPrimary, fontSize = 14.sp, fontWeight = FontWeight.Bold, letterSpacing = 1.sp)
                    Spacer(modifier = Modifier.height(10.dp))
                    Button(
                        onClick = {
                            val intent = Intent(Intent.ACTION_VIEW, Uri.parse(trailer))
                            context.startActivity(intent)
                        },
                        colors = ButtonDefaults.buttonColors(containerColor = LudexCardBg),
                        shape = RoundedCornerShape(8.dp),
                        modifier = Modifier.fillMaxWidth().height(48.dp)
                    ) {
                        Icon(Icons.Default.PlayArrow, contentDescription = null, tint = LudexAccent)
                        Spacer(modifier = Modifier.width(8.dp))
                        Text("Watch Official Trailer Video", color = LudexTextPrimary)
                    }
                }

                Spacer(modifier = Modifier.height(30.dp))
                Text(text = "PLAYER REVIEWS", color = LudexTextPrimary, fontSize = 16.sp, fontWeight = FontWeight.Bold)
                Spacer(modifier = Modifier.height(12.dp))

                Card(colors = CardDefaults.cardColors(containerColor = LudexCardBg), modifier = Modifier.fillMaxWidth(), shape = RoundedCornerShape(12.dp)) {
                    Column(modifier = Modifier.padding(12.dp)) {
                        Text("Add Your Documentation", color = LudexTextPrimary, fontSize = 12.sp, fontWeight = FontWeight.Bold)
                        Spacer(modifier = Modifier.height(8.dp))
                        Row(verticalAlignment = Alignment.CenterVertically) {
                            Text("Rating: ", color = LudexTextSecondary, fontSize = 12.sp)
                            (1..5).forEach { star ->
                                IconButton(onClick = { userRating = star }, modifier = Modifier.size(28.dp)) {
                                    Icon(
                                        Icons.Default.Star,
                                        contentDescription = null,
                                        tint = if (star <= userRating) LudexAccent else LudexTextSecondary,
                                        modifier = Modifier.size(16.dp)
                                    )
                                }
                            }
                        }
                        Spacer(modifier = Modifier.height(8.dp))
                        OutlinedTextField(
                            value = commentText,
                            onValueChange = { commentText = it },
                            placeholder = { Text("Share your experience or rating context...", color = LudexTextSecondary, fontSize = 12.sp) },
                            modifier = Modifier.fillMaxWidth().height(70.dp),
                            colors = OutlinedTextFieldDefaults.colors(focusedTextColor = LudexTextPrimary, unfocusedTextColor = LudexTextPrimary, focusedBorderColor = LudexAccent, unfocusedBorderColor = LudexTextSecondary)
                        )
                        Spacer(modifier = Modifier.height(8.dp))
                        Button(
                            onClick = {
                                val currentUid = auth.currentUser?.uid ?: return@Button
                                if (commentText.isBlank()) return@Button
                                val reviewId = db.collection("Reviews").document().id
                                val newReview = UserReview(
                                    id = reviewId,
                                    gameId = gameId,
                                    uid = currentUid,
                                    username = auth.currentUser?.email?.substringBefore("@") ?: "Elite Player",
                                    rating = userRating,
                                    comment = commentText,
                                    timestamp = Timestamp.now()
                                )
                                db.collection("Reviews").document(reviewId).set(newReview)
                                    .addOnSuccessListener {
                                        commentText = ""
                                        Toast.makeText(context, "Review Published!", Toast.LENGTH_SHORT).show()
                                    }
                            },
                            colors = ButtonDefaults.buttonColors(containerColor = LudexAccent),
                            modifier = Modifier.align(Alignment.End)
                        ) {
                            Text("Submit", color = Color.Black, fontWeight = FontWeight.Bold, fontSize = 12.sp)
                        }
                    }
                }

                Spacer(modifier = Modifier.height(16.dp))

                if (reviewsList.isEmpty()) {
                    Text("No reviews written yet. Be the first explorer!", color = LudexTextSecondary, fontSize = 12.sp, modifier = Modifier.padding(vertical = 12.dp))
                } else {
                    reviewsList.forEach { rev ->
                        Card(colors = CardDefaults.cardColors(containerColor = LudexCardBg), modifier = Modifier.fillMaxWidth().padding(vertical = 4.dp)) {
                            Column(modifier = Modifier.padding(12.dp)) {
                                Row(modifier = Modifier.fillMaxWidth(), horizontalArrangement = Arrangement.SpaceBetween) {
                                    Text(rev.username, color = LudexAccent, fontWeight = FontWeight.Bold, fontSize = 13.sp)
                                    Text("★ ${rev.rating}", color = LudexAccent, fontSize = 12.sp, fontWeight = FontWeight.Bold)
                                }
                                Spacer(modifier = Modifier.height(4.dp))
                                Text(rev.comment, color = LudexTextPrimary, fontSize = 13.sp)
                            }
                        }
                    }
                }
                Spacer(modifier = Modifier.height(40.dp))
            }
        }
    }
}
